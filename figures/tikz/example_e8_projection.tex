    \addplot[red, very thick, dashed] {cos(deg(x)) * exp(-x^2/4)};
    \addlegendentry{$\psi_2(x) = \cos(x) e^{-x^2/4}$}

    \addplot[green!70!black, very thick, dotted] {x * exp(-x^2/2)};
    \addlegendentry{$\psi_3(x) = x e^{-x^2/2}$}
