      \textcolor{green!60!black}{\checkmark} Normed division
    };

    % C: Complex numbers (dimension 2)
    \pgfmathsetmacro\yC{\ystart - \ystep}
    \node[algebra, fill=blue!15] (C) at (0, \yC) {$\mathbb{C}$ \\ \small Complex (2D)};
    \node[property, right=0.3cm of C, anchor=west] (Cprop) {
      \textcolor{red!70}{\ding{55}} Ordered field \\

    \node[algebra, fill=blue!45] (O) at (0, \yO) {$\mathbb{O}$ \\ \small Octonions (8D)};
    \node[property, right=0.3cm of O, anchor=west] (Oprop) {

    };

  \end{tikzpicture}
  \caption{Property loss cascade in the Cayley-Dickson construction. Each doubling of dimension
    (from $n$ to $2n$) results in the loss of a fundamental algebraic property. The reals
    $\mathbb{R}$ possess all properties including ordering. Complex numbers $\mathbb{C}$ lose
    ordering but retain commutativity and associativity. Quaternions $\mathbb{H}$ lose
    commutativity ($ij \neq ji$). Octonions $\mathbb{O}$ lose associativity but maintain
    alternativity (weaker property). Sedenions $\mathbb{S}$ and higher lose even normed division,
    introducing zero divisors. By Hurwitz's theorem, only the first four algebras are normed
    division algebras. This cascade fundamentally constrains the mathematical structures
    available for physics at different dimensional scales.}
  \label{fig:property-loss-cascade}
