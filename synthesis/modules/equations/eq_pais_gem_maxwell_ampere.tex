%==============================================================================
% Equation: Gravitoelectromagnetic homogeneous equations (Faraday, no-monopole)
% Source: Linearized GR, Bianchi identities in weak-field limit
% Framework: Pais | Domain: GR | Status: Theoretical
%==============================================================================
\begin{equation}
  \partial_{\mu} \tilde{F}^{G,\mu\nu} = 0
  \eqtag{P}{GR}{T}
  \label{eq:pais:gem-maxwell-homogeneous}
\end{equation}
% Notes:
%   * $\tilde{F}^{G,\mu\nu}$ dual GEM field tensor: $\tilde{F}^{G,\mu\nu} = \frac{1}{2} \epsilon^{\mu\nu\rho\sigma} F^{G}_{\rho\sigma}$
%   * $\epsilon^{\mu\nu\rho\sigma}$ Levi-Civita tensor (totally antisymmetric)
%   * 3-vector form (Faraday law): $\nabla \times \mathbf{E}_{g} = -\frac{\partial \mathbf{B}_{g}}{\partial t}$
%   * 3-vector form (no monopoles): $\nabla \cdot \mathbf{B}_{g} = 0$
%   * Analog of Maxwell equation $\partial_{\mu} \tilde{F}^{\mu\nu} = 0$ for electromagnetism
%   * These are identically satisfied if $F^{G}_{\mu\nu} = \partial_{\mu} h_{\nu} - \partial_{\nu} h_{\mu}$
