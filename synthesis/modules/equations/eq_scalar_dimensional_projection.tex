% ============================================================================
% Equation: Dimensional Projection Mapping
% ============================================================================
% Maps scalar field configurations from higher dimensions (4D-8D) to observable
% 3D space through exponential damping at characteristic length scales.
% ============================================================================

\begin{equation}
  \phi^{(d)}(x) = \sum_{i=1}^{d} \phi_i \exp\left(-\frac{2\pi r}{L_i}\right) \cos\left(\frac{2\pi x_i}{L_i} + \delta_i\right), \quad d \in \{3,4,\ldots,8\}
  \label{eq:aether:dimensional-projection}
  \eqtag{S}{MATH}{T}
\end{equation}

\noindent where:
\begin{itemize}[nosep]
  \item $\phi^{(d)}$: Scalar field in $d$ dimensions
  \item $\phi_i$: Mode amplitude for dimension $i$
  \item $r = |x|$: Radial distance from origin
  \item $L_i$: Characteristic length scale for dimension $i$
  \item $x_i$: Coordinate in dimension $i$
  \item $\delta_i$: Phase offset for dimension $i$
  \item $d$: Number of dimensions (3D through 8D)
\end{itemize}
