\begin{align}
  \ell_P &= \sqrt{\frac{\hbar G}{c^3}} \approx 1.616 \times 10^{-35} \, \text{m}
  \label{eq:planck:length} \\[6pt]
  m_P &= \sqrt{\frac{\hbar c}{G}} \approx 2.176 \times 10^{-8} \, \text{kg}
  \label{eq:planck:mass} \\[6pt]
  t_P &= \frac{\ell_P}{c} = \sqrt{\frac{\hbar G}{c^5}} \approx 5.391 \times 10^{-44} \, \text{s}
  \label{eq:planck:time} \\[6pt]
  E_P &= m_P c^2 = \sqrt{\frac{\hbar c^5}{G}} \approx 1.956 \times 10^9 \, \text{J}
  \label{eq:planck:energy} \\[6pt]
  q_P &= \sqrt{4\pi\varepsilon_0 \hbar c} \approx 1.876 \times 10^{-18} \, \text{C}
  \label{eq:planck:charge}
\end{align}
% Notes: Planck units are the natural units of quantum gravity, where
% ℏ = c = G = 1 (in geometric units).
%
% SI 2019: h and c are exact by definition, so ℏ = h/(2$\pi$) is also exact.
% Only G has measurement uncertainty (~2.2$\times$10⁻⁵ relative).
%
% These units mark the boundary where:
% - Classical GR breaks down (spacetime quantization)
% - Quantum field theory breaks down (gravity non-renormalizable)
% - New physics (string theory, loop quantum gravity, etc.) must emerge
%
% All Planck units are constructed from three fundamental constants:
%   ℏ (quantum mechanics)
%   c (special relativity)
%   G (general relativity)
