\begin{equation}
  T_{\mu\nu}^{(\phi)} = \partial_\mu\phi \partial_\nu\phi - g_{\mu\nu}\left[\frac{1}{2}g^{\alpha\beta}\partial_\alpha\phi\partial_\beta\phi + V(\phi)\right] + \xi\left(R_{\mu\nu} - \frac{1}{2}g_{\mu\nu}R\right)\phi^2
  \eqtag{S}{GR}{T}
  \label{eq:aether:stress-energy-tensor}
\end{equation}
