\begin{equation}
  \chi^2 = \sum_i\frac{[d_L(z_i; \Omega_m, \Omega_{\phi}, w_{\phi}) - d_{L,\text{obs}}(z_i)]^2}{\sigma_i^2}, \quad
  w_{\phi}(a) = w_0 + w_a(1-a)
  \eqtag{S}{COSMO}{E}
  \label{eq:cosmo:parameter-fit}
\end{equation}
% Notes: Chi-squared statistic for fitting cosmological parameters to luminosity distance
% measurements from Type Ia supernovae. The model predicts distance $d_L(z_i)$ at redshift $z_i$
% given matter density $\Omega_m$, scalar field density $\Omega_{\phi}$, and equation of state
% parameter $w_{\phi}(a)$ with scale factor $a$. The CPL parametrization (second equation)
% uses present-day value $w_0$ and evolution parameter $w_a$. Observed distances $d_{L,\text{obs}}$
% have uncertainties $\sigma_i$. Current constraints: $\Omega_{\phi} \approx 0.7$, $w_0 \approx -1$,
% $w_a \in [-0.5, 0.5]$. Aether scalar dark energy models predict specific trajectories in
% the $w_0$-$w_a$ plane testable with Euclid and Roman Space Telescope observations.
