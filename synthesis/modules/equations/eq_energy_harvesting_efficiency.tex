  \eta_{\text{ZPE}} = \frac{P_{\text{out}}}{P_{\text{in}}} = \frac{\omega_{\text{res}}Q_{\text{cav}}}{4\pi}\left(\frac{g_{\phi}\phi_0}{\omega_{\text{res}}}\right)^2\left[1 - \exp\left(-\frac{\omega_{\text{res}}\tau}{Q_{\text{cav}}}\right)\right], \quad
  P_{\text{out}} = \frac{\hbar\omega_{\text{res}}^4 V_{\text{cav}}}{2\pi^2 c^3}\alpha_{\text{enh}}
  \eqtag{S}{ENERGY}{E}
  \label{eq:energy:harvesting-efficiency}
\end{equation}
