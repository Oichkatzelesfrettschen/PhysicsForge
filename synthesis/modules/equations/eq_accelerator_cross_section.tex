  \sigma_{\text{tot}} = \sigma_{\text{SM}} + \Delta\sigma_{\phi} = \sigma_{\text{SM}}\left(1 + \frac{g_{\phi}^2 s}{(s - m_{\phi}^2)^2 + m_{\phi}^2\Gamma_{\phi}^2}\right), \quad
  s = (p_1 + p_2)^2
  \eqtag{P}{QM}{E}
  \label{eq:accelerator:cross-section}
\end{equation}
