% The modified Friedmann equations including scalar field contributions
% to the expansion rate and acceleration of the universe.
% ============================================================================

\begin{equation}
  H^2 = \frac{8\pi G}{3} \left[ \rho_m + \rho_r + \rho_\phi + \rho_{\text{ZPE}}^{\text{eff}}(\phi) \right] + \frac{k}{a^2}
  \label{eq:aether:expansion-dynamics}
  \eqtag{S}{COSMO}{T}
\end{equation}

\noindent where:
\begin{itemize}[nosep]
  \item $H = \dot{a}/a$: Hubble parameter
  \item $G$: Newton's gravitational constant
  \item $\rho_m$: Matter density
  \item $\rho_r$: Radiation density
  \item $\rho_\phi = \frac{1}{2}\dot{\phi}^2 + V(\phi)$: Scalar field energy density
  \item $\rho_{\text{ZPE}}^{\text{eff}}(\phi)$: Effective ZPE density modulated by $\phi$
  \item $k$: Spatial curvature ($k = 0, \pm 1$)
  \item $a$: Scale factor
\end{itemize}
