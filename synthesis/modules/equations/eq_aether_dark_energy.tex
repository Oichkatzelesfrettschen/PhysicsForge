% ============================================================================
% Equation: Dark Energy from Scalar Fields
% ============================================================================
% The effective dark energy density arising from time-varying scalar field
% potentials, providing a dynamical alternative to the cosmological constant.
% ============================================================================

\begin{equation}
  \rho_{\text{DE}}(\phi,t) = V_0 \left[ 1 + A \cos(\omega_H t) \right] + \frac{1}{2}\dot{\phi}^2 + \frac{1}{2}(\nabla\phi)^2
  \label{eq:aether:dark-energy}
  \eqtag{A}{COSMO}{T}
\end{equation}

\noindent where:
\begin{itemize}[nosep]
  \item $\rho_{\text{DE}}$: Dark energy density
  \item $V_0 = (10^{-3} \text{ eV})^4$: Vacuum energy scale
  \item $A = 0.1-0.3$: Oscillation amplitude
  \item $\omega_H = H_0 \sqrt{\Omega_\Lambda}$: Hubble-scale frequency
  \item $\dot{\phi}$: Time derivative of scalar field (kinetic energy)
  \item $\nabla\phi$: Spatial gradient (gradient energy)
  \item $t$: Cosmic time
\end{itemize}
