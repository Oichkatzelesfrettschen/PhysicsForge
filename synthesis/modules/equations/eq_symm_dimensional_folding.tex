\begin{equation}
  \boxed{
  \frac{d\gamma}{dt} = -\nabla V_{\text{fold}}[\gamma(t)]
  }
  \eqtag{G}{DM}{FP}
  \label{eq:module:genesis-folding-pathway}
\end{equation}

\noindent where:
\begin{itemize}[noitemsep]
  \item $\gamma: [0,1] \to \mathcal{D}$: Folding pathway in dimension space
  \item $V_{\text{fold}}$: Folding potential incorporating all forces
  \item $t$: Evolution parameter along pathway
\end{itemize}

\vspace{0.5em}
\noindent\textbf{Folding Potential:}
\begin{equation}
  V_{\text{fold}}[d] = V_{\text{grav}}[d] + V_{\text{EM}}[d] + V_{\text{weak}}[d] + V_{\text{strong}}[d] + V_{\text{fractal}}[d]
  \eqtag{G}{DM}{VP}
\end{equation}

\vspace{0.5em}
\noindent\textbf{Projection Operator:}
\begin{equation}
  \mathcal{P}_{d_1 \to d_2}[f] = \int_{x_{d_2+1}}^{x_{d_1}} f(x_1, \ldots, x_{d_1}) \prod_{i=d_2+1}^{d_1} K_i(x_i) dx_i
  \eqtag{G}{DM}{PO}
\end{equation}

\vspace{0.5em}
\noindent\textbf{Fractional Dimension Operator:}
\begin{equation}
  \mathcal{D}^{n+\alpha} = \mathcal{D}^n \cdot (1 - \alpha) + \mathcal{D}^{n+1} \cdot \alpha + \sum_{k=1}^{\infty} \frac{(-1)^k}{\varphi^k} \sin(2\pi k \alpha) \mathcal{D}^{n+k}
  \eqtag{G}{DM}{FD}
\end{equation}

\noindent where $\varphi = (1+\sqrt{5})/2$ is the golden ratio, ensuring fractal self-similarity.

\vspace{0.5em}
\noindent\textbf{Accessibility Function:}
\begin{equation}
  A_d(E) = \exp\left(-\frac{M_{\text{fold}}^d}{E}\right) \cdot \Theta(E - E_{\text{threshold}}^d)
  \eqtag{G}{DM}{AF}
\end{equation}

\noindent\textbf{Physical Significance:} Dimensional folding provides a dynamic mechanism for dimension transitions, allowing fractional dimensions and energy-dependent accessibility. This explains how extra dimensions can be hidden at low energies yet become observable at high energies.
