% ============================================================================
% Equation: Modified Casimir Force with Scalar Field
% ============================================================================
% The Casimir force between parallel plates modified by scalar field coupling,
% showing enhancement factors from direct coupling and gradient effects.
% ============================================================================

\begin{equation}
  F_{\text{Casimir}}^{(\phi)} = F_{\text{Casimir}}^{(0)} \left( 1 + \kappa \frac{\phi}{M_P} + \alpha \frac{\nabla^2\phi}{M_P \Lambda^2} \right)
  \label{eq:aether:casimir-modification}
  \eqtag{S}{QED}{T}
\end{equation}

\noindent where:
\begin{itemize}[nosep]
  \item $F_{\text{Casimir}}^{(\phi)}$: Modified Casimir force with scalar field
  \item $F_{\text{Casimir}}^{(0)} = -\pi^2\hbar c/(240 d^4)$: Standard Casimir force
  \item $\kappa = 0.15-0.25$: Direct scalar coupling coefficient
  \item $\phi$: Scalar field amplitude
  \item $M_P = 1.22 \times 10^{19}$ GeV: Planck mass
  \item $\alpha \approx 0.08$: Gradient coupling coefficient
  \item $\Lambda$: UV cutoff scale
  \item $\nabla^2\phi$: Laplacian of scalar field
\end{itemize}
