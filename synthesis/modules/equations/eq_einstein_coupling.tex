\begin{equation}
  G_{\mu\nu} = \frac{8\pi G}{c^4} T_{\mu\nu} \quad \Rightarrow \quad \text{Inverse coupling: } \frac{c^4}{G}
  \eqtag{G}{GR}{V}
  \label{eq:einstein:coupling}
\end{equation}
% Notes: Einstein's field equations relate spacetime curvature (G_$\mu$$\nu$) to
% energy-momentum (T_$\mu$$\nu$) with coupling constant 8$\pi$G/c⁴.
%

% establishing the natural force scale in general relativity.
%
% This is not phenomenological---it follows directly from the structure of
% Einstein's equations. The factor 8$\pi$ is geometric (surface area of unit
% sphere), while c⁴/G is the fundamental scale.
