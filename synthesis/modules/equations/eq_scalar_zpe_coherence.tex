% ============================================================================
% Equation: ZPE Coherence Function
% ============================================================================
% Describes the coherence of zero-point energy fluctuations under scalar field
% organization, quantifying the transition from quantum foam chaos to crystalline order.
% ============================================================================

\begin{equation}
  \mathcal{C}(\kappa, \phi) = \frac{1}{1 + e^{-\beta(\kappa - \kappa_c)}} \times \left( 1 - e^{-\phi^2/\phi_0^2} \right)
  \label{eq:aether:zpe-coherence}
  \eqtag{S}{QFT}{T}
\end{equation}

\noindent where:
\begin{itemize}[nosep]
  \item $\mathcal{C}$: Coherence function (0 = complete disorder, 1 = perfect order)
  \item $\kappa$: Foam density parameter
  \item $\kappa_c = 0.9$: Critical foam density for phase transition
  \item $\beta = 10$: Transition sharpness parameter
  \item $\phi$: Scalar field amplitude
  \item $\phi_0$: Characteristic field scale for coherence onset
\end{itemize}
