\begin{equation}
  P_{\text{total}} = P_{\text{EM}} + P_{\text{scalar}} + P_{\text{losses}} = \frac{F_T^2}{2\eta_{\text{prop}}m} + \frac{V_{\text{gen}}^2}{R_{\text{gen}}} + P_{\text{cooling}}, \quad
  F_T = \frac{\alpha_{\text{GEM}}}{c^2}\oint B_{\mu}B^{\mu}dV
  \eqtag{G}{PROP}{S}
  \label{eq:propulsion:power-budget}
\end{equation}
