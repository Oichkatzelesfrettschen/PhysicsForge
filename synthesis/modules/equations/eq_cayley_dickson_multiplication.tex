\begin{equation}
  (a,b)\cdot(c,d)
    = \left(ac - d\bar{b},\, \bar{a}d + cb\right)
  \eqtag{M}{ALG}{cd}
  \label{eq:cayley-dickson}
\end{equation}
% with $a,b,c,d \in \mathcal{A}_n$ and conjugation inherited from the prior stage.
