\begin{equation}
  \rho_{\text{dark\_matter}} = \frac{\Sigma M_{\text{baryon}}}}{V_{\text{eff}}}
  \eqtag{A}{EM}{T}
  \label{eq:aether:dark-matter-distribution}
\end{equation}
% Notes: This equation represents the effective density of dark matter, calculated
% as the sum of baryonic mass (\(\Sigma M_{\text{baryon}}\)) divided by the effective volume (\(V_{\text{eff}}\)).
