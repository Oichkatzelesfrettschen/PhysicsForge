% ============================================================================
% Equation: Scalar-ZPE Coupling
% ============================================================================
% This equation describes the coupling between scalar fields and zero-point
% energy density in the Aether framework. The nonlinear coupling enables
% modulation of vacuum energy and extraction of ZPE through scalar field control.
% ============================================================================

\begin{equation}
  \delta g_{\text{ZPE}}(\phi) = g_0 + \lambda \phi \rho_{\text{ZPE}}^2
  \label{eq:aether:scalar-zpe-coupling}
  \eqtag{S}{QFT}{T}
\end{equation}

\noindent where:
\begin{itemize}[nosep]
  \item $\delta g_{\text{ZPE}}$: Effective coupling strength modification
  \item $g_0$: Base coupling constant ($g_0 \approx 0.15$ experimentally)
  \item $\lambda$: Nonlinear coupling parameter (dimensions: [mass]$^{-5}$ in natural units)
  \item $\phi$: Scalar field amplitude
  \item $\rho_{\text{ZPE}}$: Zero-point energy density
\end{itemize}
