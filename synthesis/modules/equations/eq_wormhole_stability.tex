  \frac{d^2r_{\text{throat}}}{dt^2} = -\frac{4\pi G}{c^2}(\rho + 3p)r_{\text{throat}} + \frac{\lambda_{\phi}}{r_{\text{throat}}^3}\phi_0^2\left[1 - \left(\frac{r_c}{r_{\text{throat}}}\right)^4\right]
  \eqtag{S}{GR}{S}
  \label{eq:wormhole:stability}
\end{equation}
